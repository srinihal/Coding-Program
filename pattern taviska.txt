#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,j,n;
    cin>>n;
    for(i=n;i>=1;i--)
    {
        for(j=n;j>i;j--)
        {
            cout<<j;
        }
        for(j=1;j<=2*i-1;j++)
        {
            cout<<i;
        }
        for(j=i+1;j<=n;j++)
        {
            cout<<j;
        }
        cout<<endl;
    }
     for(i=1;i<n;i++)
    {
        for(j=n;j>i;j--)
        {
            cout<<j;
        }
        for(j=1;j<=2*i-1;j++)
        {
            cout<<i+1;
        }
        for(j=i+1;j<=n;j++)
        {
            cout<<j;
        }
        cout<<endl;
    }
}
